int check(int k){
    return k /3 * 3 == k;
}
int main(){
    int i = 0, s = 0;
    int a[10];
    while(i < 10){
        a[i] = i * (i - 5);
        i = i + 1;
    }
    i = 0;
    while(i < 10){
        if (check(a[i])){
            s = s + 1;
        }
        i = i + 1;
    }
    write(s);
    return 0;
}
